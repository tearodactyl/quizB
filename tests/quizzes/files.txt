invalid_types_quiz.json
{
  "title": "Invalid Types Quiz",
  "questions": [
    {
      "question": "What is the capital of France?",
      "options": "Paris, London, Berlin",  // Incorrect type: should be an array
      "correct": "1"  // Incorrect type: should be an integer
    }
  ]
}

exceeds_length_quiz.json
{
  "title": "Exceeds Length Quiz",
  "questions": [
    {
      "question": "A" * 2000,  // String exceeds maximum length
      "options": ["Option 1", "Option 2"],
      "correct": 1
    }
  ]
}
Note: Since JSON does not support expressions like "A" * 2000, you'll need to generate this string in code when creating the file.
